ids = {'user1': [213, 213, 213, 15, 213],
'user2': [54, 54, 119, 119, 119],
'user3': [213, 98, 98, 35]}

l = []
i = 0
j = 0
while i < len(ids):
  while j < len(ids["user"+str(i+1)]):
    if l.count(ids["user"+str(i+1)][j]) == 0:
      l.append(ids["user"+str(i+1)][j])
    j += 1
  j = 0
  i += 1
print(l)
