Vertical1 = int(input ("Введите номер клетки по вертикале, где стоит слон: "))
Horizontal1 = int(input ("Введите номер клетки по горизонтале, где стоит слон: "))
Vertical2 = int(input ("Введите номер клетки по вертикале, куда хотите поставить слона: "))
Horizontal2 = int(input ("Введите номер клетки по горизонтале, куда хотите поставить слона: "))

if ((Vertical1 - Vertical2 == Horizontal1 - Horizontal2) or (Vertical1 - Vertical2 == Horizontal2 - Horizontal1)):
  print("YES")
else:
  print("NO")
